\begin{theorem}[Moment generating functions for linear transformations]
  For any constants $a$ and $b$, the mgf of the random variable $aX + b$ is given by
  $$
  M_{aX + b}(t) = e^{bt} M_X(at)
  $$
\end{theorem}
